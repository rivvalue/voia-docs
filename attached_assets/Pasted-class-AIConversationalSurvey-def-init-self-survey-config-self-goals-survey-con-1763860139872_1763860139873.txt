class AIConversationalSurvey:
    def __init__(self, survey_config):
        self.goals = survey_config["goals"]
        self.max_questions = survey_config["max_questions"]
        self.conversation_history = []
        self.extracted_data = {}
        self.step_count = 0

    def _add_to_history(self, sender, message):
        self.conversation_history.append({
            "sender": sender,
            "message": message,
            "timestamp": time.time()
        })

    def handle_user_message(self, user_message: str):
        """
        MAIN ENTRY POINT called after each user reply.
        """

        # 1) Save user message
        self._add_to_history("User", user_message)

        # 2) Extract structured data from answer
        extracted_updates = self._extract_with_ai(user_message)
        if extracted_updates:
            self.extracted_data.update(extracted_updates)

        # Increase step (how many Q asked)
        self.step_count += 1

        # 3) Check if survey is complete (YOU decide, LLM does NOT)
        if all_goals_completed(self.goals, self.extracted_data) or \
           self.step_count >= self.max_questions:

            final_msg = (
                "Merci beaucoup pour vos réponses. "
                "Elles vont nous aider à améliorer votre expérience."
            )

            self._add_to_history("VOÏA", final_msg)

            return {
                "message": final_msg,
                "is_complete": True,
                "extracted_data": self.extracted_data,
                "conversation_history": self.conversation_history
            }

        # 4) Survey continues → find next goal
        next_goal, missing_fields = get_next_goal(self.goals, self.extracted_data)

        # 5) Ask LLM to generate the next question ONLY for that topic
        next_question = self._generate_question_with_ai(
            next_goal=next_goal,
            missing_fields=missing_fields
        )

        self._add_to_history("VOÏA", next_question)

        return {
            "message": next_question,
            "is_complete": False,
            "topic": next_goal["topic"],
            "step": self.step_count,
            "extracted_data": self.extracted_data
        }
